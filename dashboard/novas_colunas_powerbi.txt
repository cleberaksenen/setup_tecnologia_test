Faturamento_Por_Categoria = 
VAR TotalFaturamentoCategoria = 
    CALCULATE(
        SUM('setup_tecnologia vendas'[total_value]), 
        ALLEXCEPT('setup_tecnologia vendas', 'setup_tecnologia produtos'[category])
    )
RETURN
TotalFaturamentoCategoria
